pipeline {
    agent any
    stages {
        stage('Build') {
            steps {
                echo 'Building the code using Maven'
             
            }
        }
        stage('Unit and Integration Tests') {
            steps {
                echo 'Running unit and integration tests using JUnit and Selenium'
                
            }
        }
        stage('Code Analysis') {
            steps {
                echo 'Analyzing code using SonarQube'
          
            }
        }
        stage('Security Scan') {
            steps {
                echo 'Performing security scan using Checkmarx'
            
            }
        }
        stage('Deploy to Staging') {
            steps {
                echo 'Deploying to staging usingthe application to AWS EC2 instance'
               
            }
        }
        stage('Integration Tests on Staging') {
            steps {
                echo 'Performing integration tests on staging using Selenium'
              
            }
        }
        stage('Deploy to Production') {
            steps {
                echo 'Deploying the application to AWS EC2 instance for production...'
             
            }
        }
    }
    post {
        always {
            echo 'Sending email notification...'
            emailext body: "The build status is ${currentBuild.result}\n\nPlease find the build logs attached.", 
                     subject: "Jenkins build ${currentBuild.result}: ${env.JOB_NAME} (${env.BUILD_NUMBER})", 
                     attachmentsPattern: 'build.log', 
                     to: 'kingrodeski@gmail.com'
        }
    }
}
